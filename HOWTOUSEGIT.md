# Как использовать GIT

# Это создаст локальную копию репозитория на ПК
git clone https://github.com/Lun4stra/ChatService_By_DiscordEmberTeam.git


# Проверить текущие изменения
git status
# Эта команда покажет, что было добавлено или изменено.

# Подготовить изменения к коммиту
git add .
# Эта команда добавляет все изменения для фиксации.


# Сделать коммит
git commit -m "Добавил новые файлы и обновил структуру проекта"


# Отправить изменения на GitHub
git push -u origin main
# Если впервые делаем push после клонирования, ветка уже настроена на main, поэтому команда git push работает сразу.
